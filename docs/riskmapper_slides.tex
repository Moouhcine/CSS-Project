\begin{frame}{Problem}
\begin{itemize}
  \item Vulnerability scans generate many findings.
  \item We need a consistent way to \textbf{prioritize remediation}.
  \item Raw scan outputs often lack \textbf{asset context} (attack surface).
\end{itemize}
\end{frame}

\begin{frame}{Solution Overview: RiskMapper}
\begin{itemize}
  \item Desktop GUI in Python using \textbf{Flet}
  \item Uses \textbf{CVSS v3.1 Base Score} to classify risk
  \item Maps findings to \textbf{attack surface assets} (tags + exposed services)
  \item Stores everything in \textbf{SQLite} (data persists after restart)
\end{itemize}
\end{frame}

\begin{frame}{CVSS v3.1 (Base Metrics)}
\begin{itemize}
  \item Exploitability: AV, AC, PR, UI
  \item Impact: C, I, A and Scope (S)
  \item Outputs:
  \begin{itemize}
    \item Base Score (0.0--10.0) + severity class
    \item Vector string: \texttt{CVSS:3.1/AV:N/AC:L/...}
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Main Features (What the app does)}
\begin{itemize}
  \item \textbf{Calculator}: choose metrics, compute score, show severity and vector, save
  \item \textbf{Import}: CSV/JSON bulk scoring, invalid rows skipped
  \item \textbf{Assets}: asset inventory + multi-select + asset risk view (max/avg, severity counts)
  \item \textbf{Export}: global CSV export + filtered export for selected assets
\end{itemize}
\end{frame}

\begin{frame}{Workflow Demo (5 steps)}
\begin{enumerate}
  \item Create assets (hostname/IP) with tags and services
  \item Add a finding manually (Calculator) or import CSV/JSON
  \item CVSS score and severity are computed automatically
  \item Findings are linked to assets (asset name match)
  \item Export a report (all findings or selected assets CSV)
\end{enumerate}
\end{frame}

\begin{frame}{Why it matters}
\begin{itemize}
  \item Standardized risk scoring: consistent triage
  \item Faster remediation: focus on Critical/High first
  \item Attack surface view: prioritize exposed systems
  \item Persistence: SQLite keeps history across restarts
\end{itemize}
\end{frame}

\begin{frame}{Next Improvements}
\begin{itemize}
  \item SQLite enhancements: updates/editing, audit logs
  \item Import real scanner formats (Nmap XML, Nessus)
  \item Charts and trends (risk over time, SLA tracking)
  \item CVSS Temporal/Environmental scoring (advanced)
\end{itemize}
